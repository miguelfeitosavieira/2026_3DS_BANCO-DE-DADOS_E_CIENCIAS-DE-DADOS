-- QUESTÃO 3

select * from notas_fiscais;

select MATRICULA, DATA_VENDA from notas_fiscais where MATRICULA = '00237' AND DATA_VENDA = '2015-01-12';

-- Por que o CPF não é uma Chave Primária (PK) nesta tabela?
-- - O CPF não pode ser a PK porque um mesmo cliente (identificado pelo CPF) pode realizar várias compras (ter várias notas fiscais).
-- - Uma Chave Primária deve ser exclusiva para cada registro. Se o CPF fosse a PK, o cliente só poderia comprar uma única vez na vida naquela loja.
-- - Nesta tabela, a PK geralmente é o NÚMERO da nota fiscal.

select * from itens_notas_fiscais;

select CODIGO_DO_PRODUTO, QUANTIDADE from itens_notas_fiscais where QUANTIDADE >= 99;

SELECT CODIGO_DO_PRODUTO, QUANTIDADE, PRECO FROM itens_notas_fiscais WHERE QUANTIDADE >= 99 order by PRECO DESC LIMIT 1; 

-- O item com o maior valor de venda é o produto com o código 1022450, de quantidade de 99 pedidos de produtos feitos, que o orçamento total ficou por R$: 44,8542 e tudo isto você pode consultar na tabela "itens_notas_fiscais".
